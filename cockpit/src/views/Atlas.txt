<template>
  <div class="atlas-page">
    <div class="chat-header">
      <h1>🤖 Atlas AI Assistant</h1>
      <p>Natural language agent and flow creation</p>
    </div>
    <div class="chat-container">
      <div class="chat-messages" ref="messagesContainer">
        <div v-for="msg in messages" :key="msg.id" :class="['message', msg.role]">
          <div class="avatar">{{ msg.role === 'user' ? '👤' : '🤖' }}</div>
          <div class="bubble">{{ msg.content }}</div>
        </div>
        <div v-if="loading" class="message assistant">
          <div class="avatar">🤖</div>
          <div class="bubble typing">...</div>
        </div>
      </div>
      <div class="chat-input">
        <input 
          v-model="inputMessage" 
          @keyup.enter="sendMessage" 
          placeholder='Try: "create agent SupportBot" or "create flow DataPipeline" or "/help"'
        />
        <button @click="sendMessage">Send</button>
      </div>
      <div class="suggestions">
        <button v-for="s in suggestions" :key="s" @click="sendSuggestion(s)">{{ s }}</button>
      </div>
    </div>
  </div>
</template>

<script>
import api from '../services/api'

export default {
  data() {
    return {
      messages: [
        { 
          id: 1, 
          role: 'assistant', 
          content: "🤖 **Hello! I'm Atlas, your AI operations assistant.**\n\nI can help you create agents and workflows using natural language.\n\n**Examples:**\n• `create agent CustomerSupport`\n• `create flow InvoiceProcessing`\n• `list active agents`\n• `/agents`\n• `/flows`\n• `/status`\n• `/help`" 
        }
      ],
      inputMessage: '',
      loading: false,
      suggestions: ['create agent SupportBot', 'create flow DataPipeline', 'list active agents', '/help']
    }
  },
  methods: {
    async sendMessage() {
      if (!this.inputMessage.trim()) return;
      const userMsg = { id: Date.now(), role: 'user', content: this.inputMessage };
      this.messages.push(userMsg);
      const command = this.inputMessage;
      this.inputMessage = '';
      this.loading = true;
      this.$nextTick(() => this.scrollToBottom());

      try {
        const response = await this.processCommand(command);
        this.messages.push({ id: Date.now(), role: 'assistant', content: response });
      } catch (err) {
        this.messages.push({ id: Date.now(), role: 'assistant', content: `❌ Error: ${err.message || 'Something went wrong'}` });
      } finally {
        this.loading = false;
        this.scrollToBottom();
      }
    },
    
    async processCommand(text) {
      const lower = text.toLowerCase().trim();
      
      // ============ CREATE AGENT ============
      if (lower.includes('create agent')) {
        const name = text.replace(/create agent/i, '').trim();
        if (!name) return '⚠️ Please specify an agent name, e.g., "create agent SalesBot".';
        try {
          const res = await api.post('/agents', {
            name: name,
            description: `Agent created via Atlas: ${name}`,
            status: 'inactive',
            icon: '🤖'
          });
          const agent = res.data;
          return `✅ **Agent Created!**\n\nAgent "${agent.name}" has been created successfully.\n\n**Next steps:**\n• Deploy it from the Agents page\n• Add capabilities like chat or data analysis\n• Set up triggers and schedules`;
        } catch (err) {
          console.error(err);
          return `❌ Failed to create agent "${name}". Make sure the backend is running.`;
        }
      }
      
      // ============ CREATE FLOW ============
      if (lower.includes('create flow')) {
        const name = text.replace(/create flow/i, '').trim();
        if (!name) return '⚠️ Please specify a flow name, e.g., "create flow OrderProcessing".';
        try {
          const res = await api.post('/flows', {
            name: name,
            description: `Flow created via Atlas: ${name}`,
            trigger: 'manual',
            icon: '⚡'
          });
          const flow = res.data;
          return `✅ **Flow Created!**\n\nWorkflow "${flow.name}" is ready.\n\n**Next steps:**\n• Add triggers (webhook, schedule, event)\n• Add actions (API calls, notifications, data transforms)\n• Test and publish`;
        } catch (err) {
          console.error(err);
          return `❌ Failed to create flow "${name}". Make sure the backend is running.`;
        }
      }
      
      // ============ LIST ACTIVE AGENTS ============
      if (lower.includes('list active agents')) {
        try {
          const res = await api.get('/agents');
          const agents = res.data;
          const active = agents.filter(a => a.status === 'active');
          if (active.length === 0) {
            return '📋 No active agents found. Create one with "create agent [name]".';
          }
          let list = '**📋 Active Agents:**\n';
          active.forEach(a => {
            list += `• ${a.name} (${a.status})\n`;
          });
          return list;
        } catch (err) {
          return '❌ Could not fetch agents. Is the backend running?';
        }
      }
      
      // ============ /AGENTS COMMAND ============
      if (lower === '/agents' || lower === 'list agents') {
        try {
          const res = await api.get('/agents');
          const agents = res.data;
          if (agents.length === 0) {
            return '📋 No agents found. Create one with "create agent [name]".';
          }
          let list = '**📋 Your Agents:**\n';
          agents.forEach(a => {
            list += `• ${a.name} (${a.status})\n`;
          });
          return list;
        } catch (err) {
          return '❌ Could not fetch agents.';
        }
      }
      
      // ============ /FLOWS COMMAND ============
      if (lower === '/flows' || lower === 'list flows') {
        try {
          const res = await api.get('/flows');
          const flows = res.data;
          if (flows.length === 0) {
            return '📋 No flows found. Create one with "create flow [name]".';
          }
          let list = '**📋 Your Flows:**\n';
          flows.forEach(f => {
            list += `• ${f.name} (runs: ${f.executions || 0})\n`;
          });
          return list;
        } catch (err) {
          return '❌ Could not fetch flows.';
        }
      }
      
      // ============ /STATUS COMMAND ============
      if (lower === '/status') {
        try {
          const [agentsRes, flowsRes] = await Promise.all([
            api.get('/agents'),
            api.get('/flows')
          ]);
          return `📊 **System Status**\n\n• Agents: ${agentsRes.data.length}\n• Flows: ${flowsRes.data.length}\n• Status: Operational ✅`;
        } catch (err) {
          return '❌ Could not fetch system status.';
        }
      }
      
      // ============ /HELP COMMAND ============
      if (lower === '/help') {
        return `**📚 Available Commands:**\n\n• \`create agent [name]\` - Create a new AI agent\n• \`create flow [name]\` - Create a new workflow\n• \`list active agents\` - Show active agents\n• \`/agents\` - List all agents\n• \`/flows\` - List all flows\n• \`/status\` - System status\n• \`/help\` - Show this help`;
      }
      
      // ============ NATURAL LANGUAGE (uses AI) ============
      try {
        const res = await api.post('/atlas/chat', { message: text });
        return res.data.response || `🤖 I understand you want to: "${text}"\n\nTry one of these commands:\n• \`create agent [name]\`\n• \`create flow [name]\`\n• \`/help\` for more options`;
      } catch (err) {
        return `🤖 I understand you want to: "${text}"\n\nTry one of these commands:\n• \`create agent [name]\`\n• \`create flow [name]\`\n• \`/help\` for more options`;
      }
    },
    
    sendSuggestion(suggestion) {
      this.inputMessage = suggestion;
      this.sendMessage();
    },
    
    scrollToBottom() {
      const container = this.$refs.messagesContainer;
      if (container) container.scrollTop = container.scrollHeight;
    }
  }
}
</script>
