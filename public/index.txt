function loadData($file) { return file_exists($file) ? json_decode(file_get_contents($file), true) : []; }
function saveData($file, $data) { file_put_contents($file, json_encode($data, JSON_PRETTY_PRINT)); }

$dataDir = __DIR__ . '/../data';
if (!is_dir($dataDir)) mkdir($dataDir, 0777, true);
$agents = loadData(getDataFile('agents'));
$flows = loadData(getDataFile('flows'));
$memory = loadData(getDataFile('memory'));

$requestUri = strtok($_SERVER['REQUEST_URI'], '?');
$method = $_SERVER['REQUEST_METHOD'];
$input = json_decode(file_get_contents('php://input'), true) ?? [];
$path = preg_replace('#^/api/#', '/', $requestUri);
$path = rtrim($path, '/');
if (empty($path)) $path = '/';

// ==================== AGENTS ====================
if ($path === '/agents' && $method === 'GET') { echo json_encode($agents); exit; }
if ($path === '/agents' && $method === 'POST') {
    $ids = array_column($agents, 'id');
    $newId = empty($ids) ? 1 : max($ids) + 1;
    $agents[] = ['id' => $newId, 'name' => $input['name'] ?? 'New Agent', 'description' => $input['description'] ?? '', 'status' => $input['status'] ?? 'inactive', 'icon' => $input['icon'] ?? '🤖'];
    saveData(getDataFile('agents'), $agents);
    echo json_encode(end($agents)); exit;
}
if (preg_match('#^/agents/(\d+)$#', $path, $m) && $method === 'PUT') {
    $id = (int)$m[1];
    foreach ($agents as &$a) { if ($a['id'] === $id) { $a['name'] = $input['name'] ?? $a['name']; $a['description'] = $input['description'] ?? $a['description']; $a['status'] = $input['status'] ?? $a['status']; $a['icon'] = $input['icon'] ?? $a['icon']; saveData(getDataFile('agents'), $agents); echo json_encode($a); exit; } }
    http_response_code(404); echo json_encode(['error' => 'Agent not found']); exit;
}
if (preg_match('#^/agents/(\d+)$#', $path, $m) && $method === 'DELETE') {
    $id = (int)$m[1];
    foreach ($agents as $k => $a) { if ($a['id'] === $id) { unset($agents[$k]); saveData(getDataFile('agents'), array_values($agents)); http_response_code(204); exit; } }
    http_response_code(404); echo json_encode(['error' => 'Agent not found']); exit;
}
if (preg_match('#^/agents/(\d+)/run$#', $path, $m) && $method === 'POST') {
    $id = (int)$m[1];
    foreach ($agents as $a) { if ($a['id'] === $id) { $message = $input['message'] ?? 'Hello'; echo json_encode(['result' => ['response' => callLLM($message)]]); exit; } }
    http_response_code(404); echo json_encode(['error' => 'Agent not found']); exit;
}

// ==================== FLOWS ====================
if ($path === '/flows' && $method === 'GET') { echo json_encode($flows); exit; }
if ($path === '/flows' && $method === 'POST') {
    $ids = array_column($flows, 'id');
    $newId = empty($ids) ? 1 : max($ids) + 1;
    $flows[] = ['id' => $newId, 'name' => $input['name'] ?? 'New Flow', 'description' => $input['description'] ?? '', 'trigger' => $input['trigger'] ?? 'manual', 'executions' => 0, 'icon' => $input['icon'] ?? '⚡', 'config' => isset($input['config']) ? json_encode($input['config']) : null];
    saveData(getDataFile('flows'), $flows);
    echo json_encode(end($flows)); exit;
}
if (preg_match('#^/flows/(\d+)$#', $path, $m) && $method === 'PUT') {
    $id = (int)$m[1];
    foreach ($flows as &$f) { if ($f['id'] === $id) { $f['name'] = $input['name'] ?? $f['name']; $f['description'] = $input['description'] ?? $f['description']; $f['trigger'] = $input['trigger'] ?? $f['trigger']; $f['icon'] = $input['icon'] ?? $f['icon']; if (isset($input['config'])) $f['config'] = json_encode($input['config']); saveData(getDataFile('flows'), $flows); echo json_encode($f); exit; } }
    http_response_code(404); echo json_encode(['error' => 'Flow not found']); exit;
}
if (preg_match('#^/flows/(\d+)$#', $path, $m) && $method === 'DELETE') {
    $id = (int)$m[1];
    foreach ($flows as $k => $f) { if ($f['id'] === $id) { unset($flows[$k]); saveData(getDataFile('flows'), array_values($flows)); http_response_code(204); exit; } }
    http_response_code(404); echo json_encode(['error' => 'Flow not found']); exit;
}
if (preg_match('#^/flows/(\d+)/execute$#', $path, $m) && $method === 'POST') {
    $id = (int)$m[1];
    foreach ($flows as &$f) {
        if ($f['id'] === $id) {
            $f['executions']++;
            
            $config = isset($f['config']) ? (is_array($f['config']) ? $f['config'] : json_decode($f['config'], true)) : [];
            $actions = $config['actions'] ?? [];
            
            foreach ($actions as $action) {
                $type = $action['type'] ?? '';
                
                if ($type === 'slack') {
                    $webhook = $action['webhook'] ?? '';
                    $message = $action['message'] ?? 'Flow executed';
                    if ($webhook) {
                        $ch = curl_init();
                        curl_setopt($ch, CURLOPT_URL, $webhook);
                        curl_setopt($ch, CURLOPT_POST, true);
                        curl_setopt($ch, CURLOPT_POSTFIELDS, json_encode(['text' => $message]));
                        curl_setopt($ch, CURLOPT_HTTPHEADER, ['Content-Type: application/json']);
                        curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
                        curl_setopt($ch, CURLOPT_TIMEOUT, 5);
                        curl_exec($ch);
                        curl_close($ch);
                    }
                } elseif ($type === 'webhook') {
                    $url = $action['url'] ?? '';
                    $data = $action['data'] ?? [];
                    if ($url) {
                        $ch = curl_init();
                        curl_setopt($ch, CURLOPT_URL, $url);
                        curl_setopt($ch, CURLOPT_POST, true);
                        curl_setopt($ch, CURLOPT_POSTFIELDS, json_encode($data));
                        curl_setopt($ch, CURLOPT_HTTPHEADER, ['Content-Type: application/json']);
                        curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
                        curl_setopt($ch, CURLOPT_TIMEOUT, 5);
                        curl_exec($ch);
                        curl_close($ch);
                    }
                } elseif ($type === 'email') {
                    $to = $action['to'] ?? '';
                    $subject = $action['subject'] ?? 'Flow Executed';
                    $body = $action['body'] ?? 'The flow was executed successfully.';
                    $logEntry = date('Y-m-d H:i:s') . " - TO: $to | SUBJECT: $subject | BODY: $body\n";
                    file_put_contents(__DIR__ . '/../data/email_log.txt', $logEntry, FILE_APPEND);
                }
            }
            
            saveData(getDataFile('flows'), $flows);
            echo json_encode(['message' => "Flow '{$f['name']}' executed successfully"]);
            exit;
        }
    }
    http_response_code(404); echo json_encode(['error' => 'Flow not found']); exit;
}

// ==================== ATLAS CHAT ====================
if ($path === '/atlas/chat' && $method === 'POST') {
    $text = $input['message'] ?? '';
    $lower = strtolower($text);
    if (str_contains($lower, 'create agent')) {
        $name = trim(preg_replace('/create agent/i', '', $text));
        $ids = array_column($agents, 'id');
        $newId = empty($ids) ? 1 : max($ids) + 1;
        $agents[] = ['id' => $newId, 'name' => $name, 'description' => "Created via Atlas", 'status' => 'inactive', 'icon' => '🤖'];
        saveData(getDataFile('agents'), $agents);
        echo json_encode(['response' => "✅ Agent '$name' created!"]); exit;
    }
    if (str_contains($lower, 'create flow')) {
        $name = trim(preg_replace('/create flow/i', '', $text));
        $ids = array_column($flows, 'id');
        $newId = empty($ids) ? 1 : max($ids) + 1;
        $flows[] = ['id' => $newId, 'name' => $name, 'description' => "Created via Atlas", 'trigger' => 'manual', 'executions' => 0, 'icon' => '⚡'];
        saveData(getDataFile('flows'), $flows);
        echo json_encode(['response' => "✅ Flow '$name' created!"]); exit;
    }
    if (str_contains($lower, '/agents')) { $list = "Your agents:\n"; foreach ($agents as $a) $list .= "- {$a['name']} ({$a['status']})\n"; echo json_encode(['response' => $list ?: 'No agents yet.']); exit; }
    if (str_contains($lower, '/flows')) { $list = "Your flows:\n"; foreach ($flows as $f) $list .= "- {$f['name']} (runs: {$f['executions']})\n"; echo json_encode(['response' => $list ?: 'No flows yet.']); exit; }
    if (str_contains($lower, '/status')) { echo json_encode(['response' => "System Status:\n- Agents: " . count($agents) . "\n- Flows: " . count($flows)]); exit; }
    if (str_contains($lower, '/help')) { echo json_encode(['response' => "Commands:\n- create agent [name]\n- create flow [name]\n- /agents\n- /flows\n- /status\n- /help"]); exit; }
    echo json_encode(['response' => callLLM($text)]); exit;
}

// ==================== RAG ====================
if ($path === '/rag/upload' && $method === 'POST') {
    $memory[] = ['id' => uniqid(), 'content' => $input['content'] ?? '', 'metadata' => $input['metadata'] ?? [], 'created_at' => date('c')];
    saveData(getDataFile('memory'), $memory);
    echo json_encode(['message' => 'Document stored', 'id' => end($memory)['id']]); exit;
}
if ($path === '/rag/query' && $method === 'POST') {
    $query = $input['query'] ?? '';
    $results = array_filter($memory, fn($m) => stripos($m['content'], $query) !== false);
    $context = implode("\n\n", array_column($results, 'content'));
    $answer = $context ? callLLM("Context:\n$context\n\nQuestion: $query") : "I couldn't find information about '$query'.";
    echo json_encode(['answer' => $answer, 'sources' => array_values($results)]); exit;
}
