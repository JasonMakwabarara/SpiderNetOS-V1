<?php

namespace App\Services;

use Illuminate\Support\Facades\Http;
use Illuminate\Support\Facades\Log;

class LLMService
{
    protected $timeout = 30;
    protected $maxRetries = 2;
    
    protected $localEndpoints = [
        'gemma' => env('GEMMA_ENDPOINT', null),
        'qwen'  => env('QWEN_ENDPOINT', null),
    ];
    protected $openaiKey = null;

    public function __construct()
    {
        $this->openaiKey = env('OPENAI_API_KEY');
    }

    public function chat(array $messages, string $model = null)
    {
        if ($response = $this->callLocalIfAvailable($messages)) {
            return $response;
        }
        return $this->callOpenAI($messages);
    }

    protected function callLocalIfAvailable(array $messages)
    {
        $prompt = end($messages)['content'];
        
        foreach ($this->localEndpoints as $name => $endpoint) {
            if (!$endpoint) continue;
            try {
                $response = Http::timeout($this->timeout)->post($endpoint, [
                    'prompt' => $prompt,
                    'stream' => false
                ]);
                if ($response->successful()) {
                    return $response->json('response') ?? $response->body();
                }
            } catch (\Exception $e) {
                Log::warning("Local model $name failed: " . $e->getMessage());
            }
        }
        return null;
    }

    protected function callOpenAI(array $messages)
    {
        if (!$this->openaiKey) {
            return "OpenAI API key not configured.";
        }
        try {
            $response = Http::withHeaders([
                'Authorization' => 'Bearer ' . $this->openaiKey,
                'Content-Type' => 'application/json',
            ])->timeout($this->timeout)->post('https://api.openai.com/v1/chat/completions', [
                'model' => 'gpt-4o-mini',
                'messages' => $messages,
                'max_tokens' => 200,
            ]);
            if ($response->successful()) {
                return $response->json('choices.0.message.content');
            }
        } catch (\Exception $e) {
            Log::error("OpenAI failed: " . $e->getMessage());
        }
        return "AI service temporarily unavailable.";
    }
}
