// Agent Run with multi-provider LLM (local + OpenAI)
Route::post('/agents/{id}/run', function ($id, Request $request) {
    $agent = DB::table('agents')->where('id', $id)->first();
    if (!$agent) {
        return response()->json(['error' => 'Agent not found'], 404);
    }
    $message = $request->input('message', 'Hello');
    $llm = new LLMService();
    $response = $llm->chat([
        ['role' => 'system', 'content' => "You are {$agent->name}. Be helpful and concise."],
        ['role' => 'user', 'content' => $message]
    ]);
    return response()->json(['result' => ['response' => $response]]);
});

// Flow Execute with external actions (Slack, webhook)
Route::post('/flows/{id}/execute', function ($id) {
    $flow = DB::table('flows')->where('id', $id)->first();
    if (!$flow) return response()->json(['error' => 'Flow not found'], 404);
    DB::table('flows')->where('id', $id)->increment('executions');
    $config = json_decode($flow->config ?? '{}', true);
    foreach ($config['actions'] ?? [] as $action) {
        if ($action['type'] === 'slack') {
            Http::post($action['webhook'], ['text' => $action['message']]);
        } elseif ($action['type'] === 'webhook') {
            Http::post($action['url'], $action['data'] ?? []);
        }
    }
    return response()->json(['message' => "Flow executed with " . count($config['actions'] ?? []) . " actions"]);
});

// ==================== ATLAS CHAT ====================

Route::post('/atlas/chat', function (Request $request) {
    $text = $request->message;
    $lower = strtolower($text);
    
    if (str_contains($lower, 'create agent')) {
        $name = trim(preg_replace('/create agent/i', '', $text));
        $id = (string) Str::uuid();
        DB::table('agents')->insert([
            'id' => $id,
            'tenant_id' => getTenantId(),
            'name' => $name,
            'status' => 'inactive',
            'created_at' => now(),
            'updated_at' => now(),
        ]);
        return response()->json(['response' => "✅ Agent '$name' created!"]);
    }
    
    if (str_contains($lower, 'create flow')) {
        $name = trim(preg_replace('/create flow/i', '', $text));
        $id = (string) Str::uuid();
        DB::table('flows')->insert([
            'id' => $id,
            'tenant_id' => getTenantId(),
            'name' => $name,
            'executions' => 0,
            'created_at' => now(),
            'updated_at' => now(),
        ]);
        return response()->json(['response' => "✅ Flow '$name' created!"]);
    }
    
    if (str_contains($lower, '/agents')) {
        $agents = DB::table('agents')->get();
        if ($agents->isEmpty()) return response()->json(['response' => 'No agents yet.']);
        $list = "**Your agents:**\n";
        foreach ($agents as $a) $list .= "- {$a->name} ({$a->status})\n";
        return response()->json(['response' => $list]);
    }
    
    if (str_contains($lower, '/flows')) {
        $flows = DB::table('flows')->get();
        if ($flows->isEmpty()) return response()->json(['response' => 'No flows yet.']);
        $list = "**Your flows:**\n";
        foreach ($flows as $f) $list .= "- {$f->name} (runs: {$f->executions})\n";
        return response()->json(['response' => $list]);
    }
    
    // If no command, use LLM
    $llm = new LLMService();
    $response = $llm->chat([
        ['role' => 'system', 'content' => 'You are Atlas, an AI operations assistant. Be concise and helpful.'],
        ['role' => 'user', 'content' => $text]
    ]);
    return response()->json(['response' => $response]);
});

// ==================== RAG (Retrieval-Augmented Generation) ====================

Route::post('/rag/upload', function (Request $request) {
    $id = (string) Str::uuid();
    DB::table('memory_nodes')->insert([
        'id' => $id,
        'tenant_id' => getTenantId(),
        'content' => $request->input('content'),
        'metadata' => json_encode($request->input('metadata', [])),
        'created_at' => now(),
        'updated_at' => now(),
    ]);
    return response()->json(['message' => 'Document stored', 'id' => $id]);
});

Route::post('/rag/query', function (Request $request) {
    $query = $request->input('query');
    $tenantId = getTenantId();
    $results = DB::table('memory_nodes')
        ->where('tenant_id', $tenantId)
        ->where('content', 'LIKE', '%' . $query . '%')
        ->get();
    $context = $results->pluck('content')->implode("\n\n");
    $llm = new LLMService();
    $answer = $llm->chat([
        ['role' => 'system', 'content' => 'Answer based on context. If not found, say "I don\'t know".'],
        ['role' => 'user', 'content' => "Context:\n$context\n\nQuestion: $query"]
    ]);
    return response()->json(['answer' => $answer, 'sources' => $results]);
});
